Ciclos y estructuras de control

Estructura if

Python

alumnos = 30
if alumnos <= 5:
    print("La clase no se abrira")
else:
    print("Todo está bien")

Julia

alumnos = 30
if alumnos <= 5
    print("La clase no se abrira")
else
    print("Todo está bien")
end

Ciclo While

Python

c = 1
itera = 8
while c <= itera:
    c = c +1
    print("Vamos en la iteracion",end="")
    print(c)
print("Salimos")

Julia

c = 1
itera = 8
while c <= itera
    global c
    c = c +1
    print("Vamos en la iteracion",end="")
    print(c)
end
print("Salimos")

Ciclo For

Python

suma = 0
for i in range(1,6):
    suma = suma + i
print(suma)

Julia

suma = 0
for i in range(1,6):
    global suma
    suma = suma + i
end
